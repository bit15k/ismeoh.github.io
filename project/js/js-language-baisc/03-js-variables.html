                                      # JS变量

                                      ECMAScript 变量是松散类型的，意思是变量可以用于保存任何类型的数据。每个变量只不过是一个用于保存任意值的命名占位符。

                                      有 3 个关键字可以声明变量：var、const 和 let。其中，var 在 ECMAScript 的所有版本中都可以使用，而 const 和 let 只能在 ECMAScript 6 及更晚的版本中使用。

                                      # var关键字

                                      主要包含

                                      • var的作用域
                                      • var的声明提升

                                      var message
                                      上述var表示关键字，message表示变量标识符或变量名
                                      表示定义一个变量message可以保留任何类型变量

                                      变量的undefined
                                      一个变量只是上面定义了，但并没有初始化则其值是undefined

                                      变量的初始化
                                      所谓初始化是指变量定义和赋值同时进行如 var message = 1
                                      不推荐的写法message = 1即不指明关键字

                                      同时初始化多个写法var message = "msg",found = false, age = 1

                                      # var作用域

                                      函数内部使用var关键字声明的变量，是局部变量，函数退出时候会销毁

                                      function test() { 
 var message = "hi"; // 局部变量
} 
test(); 
console.log(message); // 出错！

                                      若要在函数外部能够访问，可以在函数内部定义全局变量

                                      var message;
function test() { 
 message = "hi"; 
} 
test(); 
console.log(message);

                                      # var声明提升

                                      即未声明先使用在一定情况下是可以，会自动帮你提前声明

                                      函数内部提升至函数头部
                                      即可以先使用不声明，会自动帮你在头部声明

                                      function foo() { 
 console.log(age); // undefined
 var age = 26; 
} 
foo();

                                      等同于提升到头部

                                      function foo() { 
 var age;
 console.log(age); // undefined
 var age = 26; 
} 
foo();

                                      # let关键字

                                      主要包含如下内容

                                      • let是块作用域
                                      • 不会提升
                                      • 暂时性锁区
                                      • 不会成为window对象的属性
                                      • 不允许重复声明
                                      • 不建议使用条件式声明
                                      • 对比使用var和let进行for循环区别

                                      # 块作用域

                                      与var的区别如下
                                      第一个区别:let是块作用域

                                      let 声明的范围是块作用域，而 var 声明的范围是函数作用域。

                                      所谓块作用域是指的{}包裹起来的，如if,for后面的{}
                                      所谓函数作用域是指函数块作用域{}

                                      例如

                                      if (true) { 
 var name = 'Matt'; 
 console.log(name); // Matt 
} 
console.log(name); // Matt 
----------------------------
if (true) { 
 let age = 26; 
 console.log(age); // 26 
} 
console.log(age); // ReferenceError: age 没有定义

                                      下面例子说明，var关键字是函数作用域

                                      <script>
    function sayName(){
        var name = 'Nicholas'; 
        console.log(name); // 'Nicholas' 
        if (true) { 
        var name = 'Matt'; 
        console.log(name); // 'Matt' 
        } 
        if (true) { 
        var name = 'Tom'; 
        console.log(name); // 'Tom' 
        } 
        // name is????
        console.log(name); //Tom
    
}
sayName();

                                      上面，因为let是块作用域，所以在外面访问不到，但是var是极限是函数块作用域所以在外面可以访问
                                      所以函数块作用域与块作用域显然前者更大后者范围更小，块作用域是函数块作用域的子集

                                      let块作用域内冗余声明不被允许,如下

                                      var name; 
var name; 
let age; 
let age; // SyntaxError；标识符 age 已经声明过了

                                      # 不会提升

                                      第二个区别:let声明变量不会被提升,如下

                                      <script>
    // age 不会被提升
    console.log(age); // ReferenceError：age 没有定义
    let age = 26; 

</script>

                                      # 暂时性锁区

                                      上面如果是var那么会出现变量提升，但是若紧跟let，JavaScript引擎会关注到，因此在此之前不能有任何的引用，这个叫做暂时性锁区，从而导致在声明之前任何使用都会抛出错误

                                      # 不会成为window的属性

                                      var声明的变量会成为 window对象属性
                                      let 不会

                                      var name = 'Matt'; 
console.log(window.name); // 'Matt' 
let age = 26; 
console.log(window.age); // undefined

                                      # 不允许重复声明变量

                                      let age = 26; 
let age = 33;

                                      上面，由于let是块级变量只在当前行，所以不会提升，但是使用var就不同了，如下

                                      var name = 'Nicholas'; 
var name = 'Tom';
console.log(name); // tom

                                      上面，var对于重复声明的处理是，提升覆盖之前的提升，所以允许重复声明（不推荐）

                                      # 条件式声明

                                      所谓条件式声明是指，用if来判断是否定义变量，若没有在后面{}定义一个，如下

                                      if (typeof name === 'undefined') { 
 let name; 
 } 
 // name 被限制在 if {} 块的作用域内
 // 因此这个赋值形同全局赋值
  name = 'Matt'; 


 try { 
 console.log(age); // 如果 age 没有声明过，则会报错
 } 
 catch(error) { 
 let age; 

 } 
 // age 被限制在 catch {}块的作用域内
 // 因此这个赋值形同全局赋值
age = 26

                                      上面，由于let是块级作用域，所以只在if{}块中起作用，故下面赋值就不是if{}声明的变量，而不加关键字的声明等同于var声明，并且提升至头部，所以它是全局变量
                                      由于是块级作用域，所以即使try catch也无法捕获

                                      # 与for循环

                                      迭代作用域仅限循环内部

                                      for (let i = 0; i < 5; ++i) { 
 // 循环逻辑
} 
console.log(i); // ReferenceError: i 没有定义

                                      但是var声明的，会渗透外部

                                      for (var i = 0; i < 5; ++i) { 
 // 循环逻辑 
} 
console.log(i); // 5

                                      # const声明

                                      主要有如下

                                      • 与let基本一致
                                      • 必须声明变量同时初始化变量
                                      • 变量不能被修改

                                      # 与let类似

                                      const age = 26; 
age = 36; // TypeError: 给常量赋值
3.3 变量 29 
// const 也不允许重复声明
const name = 'Matt'; 
const name = 'Nicholas'; // SyntaxError 
// const 声明的作用域也是块
const name = 'Matt'; 
if (true) { 
 const name = 'Nicholas'; 
} 
console.log(name); // Matt

                                      # const与对象

                                      const若声明的是一个对象，那么可以修改对象内部的值,因为对象是引用类型，const声明的变量再堆内存种记录地址，这个地址指向了栈实际存储对象的地方，所以修改对象内部属性不受限制，因为地址没变

                                      const person = {}; 
person.name = 'Matt'; // ok

                                      # const与循环

                                      const与for-of循环同时使用没有意义,但是可以用在for-in循中

                                      let i = 0; 
for (const j = 7; i < 5; ++i) { 
 console.log(j); 
} 
// 7, 7, 7, 7, 7 
for (const key in {a: 1, b: 2}) { 
 console.log(key); 
} 
// a, b
for (const value of [1,2,3,4,5]) { 
 console.log(value); 
} 
// 1, 2, 3, 4, 5

                                      # 最佳实践

                                      • 不使用var
                                      • const优先，let次之
